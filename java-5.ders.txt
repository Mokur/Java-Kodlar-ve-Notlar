public class Ornek12 {

	public static void main(String[] args) {
		// küçükten büyüðe sýralama(sorting seletion algorithm)
		
		
		int[] dizi = {5,70,9,140,3,74,0};
		
		int gecici = 0;
		
		
		//tüm elemanlarý tek tek gezebilmek için ürettik
		for (int i = 0; i < dizi.length; i++) {
			
			//bir önceki deðerle bir sonraki deðeri her zaman kýyaslayabilmek için
			for (int j = i+1; j < dizi.length; j++) {
				if (dizi[i]>dizi[j]) {
					gecici = dizi[i];
					dizi[i] = dizi[j];
					dizi[j] = gecici;
				}
			}
		}
		
		for (int i = 0; i < dizi.length; i++) {
			System.out.print(dizi[i] +" ");
		}
		
		
		

	}

}

public class Ornek12 {

	public static void sirala(int[] dizi, char ch) {
		//int[] dizi = {5,70,9,140,3,74,0};
		
		int gecici;
		//tüm elemanlarý tek tek gezebilmek için ürettik
		for (int i = 0; i < dizi.length; i++) {
			//bir önceki deðerle bir sonraki deðeri her zaman kýyaslayabilmek için
			for (int j = i+1; j < dizi.length; j++) {
				if('>' == ch) {
					if(dizi[i]>dizi[j]) {
						gecici = dizi[i];
						dizi[i] = dizi[j];
						dizi[j] = gecici;
					}
				}else if('<' == ch) {
					if(dizi[i]<dizi[j]) {
						gecici = dizi[i];
						dizi[i] = dizi[j];
						dizi[j] = gecici;
					}
				}
			}
		}
		
		for (int i = 0; i < dizi.length; i++) {
			System.out.print(dizi[i]+" ");
		}
	}

	public static void main(String[] args) {
		// Küçükten büyüðe sýralama (Sorting Selection Algorithm)
		
		// 0 5 70 9 14 7
		// 0 5 7 9 14 70
		
		int[] dizi = {5,70,9,140,3,74,0};
		sirala(dizi, '>');
		System.out.println("\n");
		sirala(dizi, '<');
		
		
		
	}
}
